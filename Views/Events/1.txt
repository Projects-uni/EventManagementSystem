                <input type="hidden" id="editTaskId" />

                <div class="form-group">
                    <label>Durum</label>
                    <select id="editTaskStatus">
                        <option value="Pending">Beklemede</option>
                        <option value="In Progress">Devam Ediyor</option>
                        <option value="Done">Tamamlandƒ±</option>
                    </select>
                </div>

                <div class="form-group">
                    <label>Yorum</label>
                    <textarea id="editTaskComment"></textarea>
                </div>

                <div class="form-actions">
                    <button type="button" class="btn" onclick="closeEditTaskModal()" style="background:#ccc;">ƒ∞ptal</button>
                    <button type="submit" class="btn btn-success">Kaydet</button>
                </div>
            </form>
        </div>
    </div>

    <script>
        const eventId = '@Model.Id';
        const canEdit = @(canEdit ? "true" : "false");
        const userRole = '@(ViewBag.UserRole ?? "")';
        const currentUserIdStr = '@(currentUserId?.ToString() ?? "")'.toLowerCase();
        let tasksCache = [];

        function switchTab(tabName, btn) {
            document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
            document.querySelectorAll('.tab-content').forEach(content => content.classList.remove('active'));

            btn.classList.add('active');
            document.getElementById(tabName + '-tab').classList.add('active');

            if (tabName === 'tasks') loadTasks();
            if (tabName === 'participants') loadParticipants();
        }

        function openEditTaskModal(taskId) {
            const task = tasksCache.find(t => t.id === taskId);
            if (!task) return;

            document.getElementById('editTaskId').value = taskId;
            document.getElementById('editTaskStatus').value = task.status || 'Pending';
            document.getElementById('editTaskComment').value = task.comment || '';

            document.getElementById('editTaskModal').classList.add('active');
        }

        function closeEditTaskModal() {
            document.getElementById('editTaskModal').classList.remove('active');
        }

        async function submitEditTask(e) {
            e.preventDefault();

            const id = document.getElementById('editTaskId').value;
            const status = document.getElementById('editTaskStatus').value;
            const comment = document.getElementById('editTaskComment').value;

            try {
                const resp = await fetch('/Events/UpdateTask', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ id, status, comment })
                });

                const data = await resp.json();

                if (!resp.ok) {
                    alert(data.error || 'G√∂rev g√ºncellenirken hata olu≈ütu');
                    return;
                }

                closeEditTaskModal();
                loadTasks();
            } catch (err) {
                console.error('G√∂rev g√ºncelleme hatasƒ±:', err);
                alert('G√∂rev g√ºncellenirken hata olu≈ütu');
            }
        }

        async function loadTasks() {
            try {
                const response = await fetch(`/Events/GetTasks/${eventId}`);
                const tasks = await response.json();
                tasksCache = tasks;

                const container = document.getElementById('tasks-container');
                if (tasks.length === 0) {
                    container.innerHTML = '<div class="empty-state"><h4>Hen√ºz g√∂rev yok</h4><p>ƒ∞lk g√∂revinizi ekleyerek ba≈ülayƒ±n!</p></div>';
                    return;
                }

                container.innerHTML = tasks.map(task => {
                    const assignedTo = task.assignedTo || task.assignedTo;
                    const canEditTask =
                        userRole === 'Admin' ||
                        userRole === 'Organizer' ||
                        (currentUserIdStr && assignedTo && assignedTo.toLowerCase() === currentUserIdStr);

                    const priorityMap = {
                        'Low': 'D√º≈ü√ºk',
                        'Medium': 'Orta',
                        'High': 'Y√ºksek',
                        'Critical': 'Kritik'
                    };

                    const statusMap = {
                        'Pending': 'Beklemede',
                        'In Progress': 'Devam Ediyor',
                        'Done': 'Tamamlandƒ±'
                    };

                    return `
                        <div class="task-card">
                            <div class="task-header">
                                <div class="task-title">${task.name}</div>
                                <span class="priority-badge priority-${task.priority.toLowerCase()}">${priorityMap[task.priority] || task.priority}</span>
                            </div>
                            ${task.description ? `<p class="task-info">${task.description}</p>` : ''}
                            ${task.dueDate ? `<div class="task-info">üìÖ Son Tarih: ${new Date(task.dueDate).toLocaleString('tr-TR')}</div>` : '<div class="task-info">üìÖ Son tarih yok</div>'}
                            ${task.budget ? `<div class="task-info">üí∞ B√ºt√ße: ‚Ç∫${parseFloat(task.budget).toFixed(2)}</div>` : ''}
                            ${task.comment ? `<div class="task-info">üí¨ ${task.comment}</div>` : ''}
                            <div class="task-info">üë§ Atanan: ${task.assignedUser?.username || 'Atanmadƒ±'}</div>
                            <div class="task-info">Durum: <strong>${statusMap[task.status] || task.status}</strong></div>
                            ${canEditTask ? `
                                <div class="task-info" style="margin-top:8px;">
                                    <button class="btn btn-primary btn-small" onclick="openEditTaskModal('${task.id}')">G√∂revi D√ºzenle</button>
                                </div>` : ''}
                        </div>
                    `;
                }).join('');
            } catch (error) {
                console.error('G√∂revler y√ºklenirken hata:', error);
            }
        }

        async function loadParticipants() {
            try {
                const response = await fetch(`/Events/GetParticipants/${eventId}`);
                const participants = await response.json();

                const container = document.getElementById('participants-container');
                if (participants.length === 0) {
                    container.innerHTML = '<div class="empty-state"><h4>Hen√ºz katƒ±lƒ±mcƒ± yok</h4><p>ƒ∞lk katƒ±lƒ±mcƒ±nƒ±zƒ± davet edin!</p></div>';
                    return;
                }

                container.innerHTML = participants.map(p => {
                    const userObj = p.user ?? p.User;
                    const username = userObj?.username ?? userObj?.Username ?? 'Bilinmeyen Kullanƒ±cƒ±';
                    const email = userObj?.email ?? userObj?.Email ?? 'Yok';

                    return `
                        <div class="participant-card">
                            <div class="participant-info">
                                <div class="participant-details">
                                    <h4>${username}</h4>
                                    <div class="participant-meta">
                                        üìß ${email}
                                    </div>
                                </div>
                            </div>
                        </div>
                    `;
                }).join('');
            } catch (error) {
                console.error('G√∂rev Alanlar y√ºklenirken hata:', error);
            }
        }

        async function loadAssignToDropdown() {
            try {
                const resp = await fetch(`/Events/GetParticipants/${eventId}`);
                const participants = await resp.json();

                const dropdown = document.getElementById("assignToSelect");
                dropdown.innerHTML = `<option value="">-- Atanmadƒ± --</option>`;

                participants.forEach(p => {
                    const userId = p.userId ?? p.UserId;
                    const userObj = p.user ?? p.User;
                    const username = userObj?.username ?? userObj?.Username;

                    if (userId && username) {
                        dropdown.innerHTML += `
                            <option value="${userId}">
                                ${username}
                            </option>`;
                    }
                });
            } catch (err) {
                console.error("Atama listesi y√ºklenirken hata:", err);
            }
        }

        function openAddTaskModal() {
            loadAssignToDropdown();
            document.getElementById('taskModal').classList.add('active');
        }

        function closeTaskModal() {
            document.getElementById('taskModal').classList.remove('active');
            document.getElementById('taskForm').reset();
        }

        function openAddParticipantModal() {
            document.getElementById('participantModal').classList.add('active');
        }

        function closeParticipantModal() {
            document.getElementById('participantModal').classList.remove('active');
            document.getElementById('inviteUsername').value = '';
        }

        function openInviteModal() {
            document.getElementById('inviteModal').classList.add('active');
        }

        function closeInviteModal() {
            document.getElementById('inviteModal').classList.remove('active');
        }

        async function submitTask(e) {
            e.preventDefault();
            const formData = new FormData(e.target);

            const task = {
                eventId: eventId,
                name: formData.get('name'),
                description: formData.get('description') || null,
                dueDate: formData.get('dueDate') || null,
                priority: formData.get('priority'),
                budget: formData.get('budget') ? parseFloat(formData.get('budget')) : null,
                comment: formData.get('comment') || null,
                assignedTo: formData.get('assignedTo') || null,
                status: 'Pending'
            };

            try {
                const response = await fetch('/Events/AddTask', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(task)
                });

                if (response.ok) {
                    closeTaskModal();
                    loadTasks();
                } else {
                    const data = await response.json().catch(() => null);
                    alert(data?.error || 'G√∂rev olu≈üturulurken hata olu≈ütu');
                }
            } catch (error) {
                console.error('Hata:', error);
                alert('G√∂rev olu≈üturulurken hata olu≈ütu');
            }
        }

        async function inviteUserByUsername() {
            const username = document.getElementById("inviteUsername").value.trim();
            if (!username) {
                alert("L√ºtfen bir kullanƒ±cƒ± adƒ± girin.");
                return;
            }

            try {
                const resp = await fetch("/Events/AddParticipant", {
                    method: "POST",
                    headers: { "Content-Type": "application/json" },
                    body: JSON.stringify({ eventId, username })
                });

                const data = await resp.json();

                if (!resp.ok) {
                    alert(data.error || "Kullanƒ±cƒ± davet edilirken hata olu≈ütu.");
                    return;
                }

                alert("Kullanƒ±cƒ± davet edildi!");
                closeParticipantModal();
                loadParticipants();
                loadAssignToDropdown();
            } catch (err) {
                console.error(err);
                alert("Beklenmeyen bir hata olu≈ütu.");
            }
        }

        // Initial load
        loadTasks();
        loadParticipants();
    </script>
